Quantum Basics
#01 Bell State basics
Two entangled qubits. Always 00 or 11. Quantum Hello World.
-- Bell State: quantum entanglement
let q = qubits(2)
H(q[0]); CNOT(q[0], q[1])
let r = measure_all(q, shots=1000)
print(r.histogram)
-- {"00": 503, "11": 497}
#02 Superposition basics
Single qubit equally between 0 and 1. H gate creates this state.
let q = qubits(1)
H(q[0])  -- |0⟩ → (|0⟩+|1⟩)/√2
let r = measure_all(q, shots=1000)
print(r.histogram) -- {"0":~500, "1":~500}
#03 Bit Flip (X Gate) basics
X gate flips |0⟩ to |1⟩. Quantum NOT gate.
let q = qubits(1)
X(q[0])  -- |0⟩ → |1⟩
let r = measure_all(q, shots=100)
print(r.histogram) -- {"1": 100}
#04 Phase Flip (Z Gate) basics
Z gate flips the phase of |1⟩. Invisible to measurement, crucial for interference.
let q = qubits(1)
H(q[0]); Z(q[0]); H(q[0])
let r = measure_all(q, shots=100)
print(r.histogram) -- {"1": 100} (HZH = X)

#22 Teleportation algo
Transfer qubit state from Alice to Bob using Bell pair.
def teleport(angle):
    let q = qubits(3)
    Ry(q[0], angle)
    H(q[1]); CNOT(q[1], q[2])
    CNOT(q[0], q[1]); H(q[0])
    let m0 = measure(q[0])
    let m1 = measure(q[1])
    if m1 == 1: X(q[2])
    if m0 == 1: Z(q[2])
    print(f"theta={round_to(angle,3)} teleported OK")
for a in [0, PI/4, PI/2, PI]: teleport(a)

#24 Quantum RNG algo
True randomness from quantum measurement.
def qrand(lo, hi):
    let n = n_qubits_for(hi-lo+1)
    let q = qubits(n); H_all(q)
    let bits = [measure(q[i]) for i in range(n)]
    let v = bits_to_int(join([str(b) for b in bits],""))
    if v >= hi-lo+1: return qrand(lo,hi)
    return lo+v
print([qrand(1,100) for _ in range(10)])

#46 Quantum Coin prac
Quantum coin flip — true randomness.
def qcoin():
    let q = qubits(1); H(q[0])
    return "Heads" if measure(q[0])==0 else "Tails"
print([qcoin() for _ in range(10)])
#47 Dice Roll prac
Quantum dice using rejection sampling.
def qdie():
    let q = qubits(3); H_all(q)
    let v = bits_to_int(join([str(measure(q[i])) for i in range(3)],""))
    if v >= 6: return qdie()
    return v+1
print([qdie() for _ in range(10)])
